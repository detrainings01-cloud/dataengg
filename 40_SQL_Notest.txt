CREATE TABLE [IF NOT EXISTS ] ORDERS (
	ORDER_ID INTEGER  
	, CUSTOMER_ID  INTEGER 
	, PRODUCT_ID INTEGER 
	, PRICE   NUMERIC(10,2)
	, QUANTITY  INTEGER 
);

CREATE OR REPLACE TABLE PRODUCTS(
	PRODUCT_ID INTEGER 
	, PRODUCT_NAME VARCHAR(250) 
	, CATEGORY VARCHAR(250)
);

SELECT COUNT(*) FROM ORDERS ; -- TOTAL RECORDS IN THE TABLE

-- GROUP BY IMPLEMENTATION
